# NCBI-Scoring-Matrix-Converter
An Application that reads the NCBI Scoring Matrix format and convert it to a usable C# Class.

### For Example

internal class Blosum50
{
	private readonly Dictionary<char, int> _symbols = new Dictionary<char, int>

	private const int WildMistmatch = -5, WildMatch = 1;

	public override int Match(char a, char b)
	{
		int index1, index2;
		if (_symbols.TryGetValue(a, out index1) && _symbols.TryGetValue(b, out index2))
		{
			return _matrix[index1, index2];
		}
		else
		{
			return a == b ? WildMatch : WildMistmatch;
		}
	}
